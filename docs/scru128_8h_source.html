31/**
32 * Represents a SCRU128 ID and provides various converters and comparison
33 * operators.
34 */
35typedef struct Scru128Id {
36 /**
37 * Internal 128-bit byte array representation.
38 *
39 * @private
40 */
41 uint8_t _bytes[16];
42} Scru128Id;

44/** Status code returned by `scru128_generator_last_status()` function. */
45typedef enum Scru128GeneratorStatus {
46 /** Indicates that the generator has yet to generate an ID. */
47 SCRU128_GENERATOR_STATUS_NOT_EXECUTED = 0,
48
49 /**
50 * Indicates that the latest `timestamp` was used because it was greater
51 * than the previous one.
52 */
53 SCRU128_GENERATOR_STATUS_NEW_TIMESTAMP,
54
55 /**
56 * Indicates that `counter_lo` was incremented because the latest
57 * `timestamp` was no greater than the previous one.
58 */
59 SCRU128_GENERATOR_STATUS_COUNTER_LO_INC,
60
61 /**
62 * Indicates that `counter_hi` was incremented because `counter_lo` reached
63 * its maximum value.
64 */
65 SCRU128_GENERATOR_STATUS_COUNTER_HI_INC,
66
67 /**
68 * Indicates that the previous `timestamp` was incremented because
69 * `counter_hi` reached its maximum value.
70 */
71 SCRU128_GENERATOR_STATUS_TIMESTAMP_INC,
72
73 /**
74 * Indicates that the monotonic order of generated IDs was broken because
75 * the latest `timestamp` was less than the previous one by ten seconds or
76 * more.
77 */
78 SCRU128_GENERATOR_STATUS_CLOCK_ROLLBACK,
79
80 /** Indicates that the previous generation failed. */
81 SCRU128_GENERATOR_STATUS_ERROR
82} Scru128GeneratorStatus;

126/**
127 * Creates a SCRU128 ID object from field values.
128 *
129 * @param out Location where the new object is stored.
130 * @param timestamp 48-bit `timestamp` field value.
131 * @param counter_hi 24-bit `counter_hi` field value.
132 * @param counter_lo 24-bit `counter_lo` field value.
133 * @param entropy 32-bit `entropy` field value.
134 * @return Zero on success or a non-zero integer if any argument is out of the
135 * value range of the field.
136 */
137static inline int scru128_from_fields(Scru128Id *out, uint64_t timestamp,
138 uint32_t counter_hi, uint32_t counter_lo,
139 uint32_t entropy) {
140 if (timestamp > 0xffffffffffff || counter_hi > 0xffffff ||
141 counter_lo > 0xffffff) {
142 return -1;
143 }
144
145 out->_bytes[0] = timestamp >> 40;
146 out->_bytes[1] = timestamp >> 32;
147 out->_bytes[2] = timestamp >> 24;
148 out->_bytes[3] = timestamp >> 16;
149 out->_bytes[4] = timestamp >> 8;
150 out->_bytes[5] = timestamp;
151 out->_bytes[6] = counter_hi >> 16;
152 out->_bytes[7] = counter_hi >> 8;
153 out->_bytes[8] = counter_hi;
154 out->_bytes[9] = counter_lo >> 16;
155 out->_bytes[10] = counter_lo >> 8;
156 out->_bytes[11] = counter_lo;
157 out->_bytes[12] = entropy >> 24;
158 out->_bytes[13] = entropy >> 16;
159 out->_bytes[14] = entropy >> 8;
160 out->_bytes[15] = entropy;
161 return 0;
162}

164/**
165 * Creates a SCRU128 ID object from a byte array that represents a 128-bit
166 * unsigned integer.
167 *
168 * @param out Location where the new object is stored.
169 * @param bytes 16-byte byte array that represents a 128-bit unsigned integer in
170 * the big-endian (network) byte order.
171 * @return Zero on success or a non-zero integer on failure.
172 */
173static inline int scru128_from_bytes(Scru128Id *out, const uint8_t *bytes) {
174 for (int_fast8_t i = 0; i < 16; i++) {
175 out->_bytes[i] = bytes[i];
176 }
177 return 0;
178}

191/** Returns the 48-bit `timestamp` field value of a SCRU128 ID. */
192static inline uint64_t scru128_timestamp(const Scru128Id *id) {
193 return (uint64_t)id->_bytes[0] << 40 | (uint64_t)id->_bytes[1] << 32 |
194 (uint64_t)id->_bytes[2] << 24 | (uint64_t)id->_bytes[3] << 16 |
195 (uint64_t)id->_bytes[4] << 8 | (uint64_t)id->_bytes[5];
196}
197
198/** Returns the 24-bit `counter_hi` field value of a SCRU128 ID. */
199static inline uint32_t scru128_counter_hi(const Scru128Id *id) {
200 return (uint32_t)id->_bytes[6] << 16 | (uint32_t)id->_bytes[7] << 8 |
201 (uint32_t)id->_bytes[8];
202}
203
204/** Returns the 24-bit `counter_lo` field value of a SCRU128 ID. */
205static inline uint32_t scru128_counter_lo(const Scru128Id *id) {
206 return (uint32_t)id->_bytes[9] << 16 | (uint32_t)id->_bytes[10] << 8 |
207 (uint32_t)id->_bytes[11];
208}
209
210/** Returns the 32-bit `entropy` field value of a SCRU128 ID. */
211static inline uint32_t scru128_entropy(const Scru128Id *id) {
212 return (uint32_t)id->_bytes[12] << 24 | (uint32_t)id->_bytes[13] << 16 |
213 (uint32_t)id->_bytes[14] << 8 | (uint32_t)id->_bytes[15];
214}

239/**
240 * Returns a negative integer, zero, or positive integer if `lhs` is less than,
241 * equal to, or greater than `rhs`, respectively.
242 */
243static inline int scru128_compare(const Scru128Id *lhs, const Scru128Id *rhs) {
244 for (int_fast8_t i = 0; i < 16; i++) {
245 if (lhs->_bytes[i] != rhs->_bytes[i]) {
246 return lhs->_bytes[i] < rhs->_bytes[i] ? -1 : 1;
247 }
248 }
249 return 0;
250}

260/** Initializes a generator struct `g`. */
261static inline void scru128_initialize_generator(Scru128Generator *g) {
262 g->_timestamp = 0;
263 g->_counter_hi = 0;
264 g->_counter_lo = 0;
265 g->_ts_counter_hi = 0;

271 * involved in the last generation of ID.
